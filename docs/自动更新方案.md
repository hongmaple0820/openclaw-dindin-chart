## 自动更新机制

### 方案 A：Git Pull + 自动重启（推荐）

**优势：**
- ✅ 简单可靠
- ✅ 无需额外工具
- ✅ 支持回滚

**实现：**

```bash
#!/bin/bash
# auto-update.sh

REPO_DIR="$HOME/.openclaw/openclaw-dindin-chart"
LOG_FILE="$HOME/.openclaw/logs/auto-update.log"
BACKUP_DIR="$HOME/.openclaw/backups"

log() {
  echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a "$LOG_FILE"
}

# 1. 检查是否有更新
check_updates() {
  cd "$REPO_DIR"
  git fetch origin master
  
  LOCAL=$(git rev-parse HEAD)
  REMOTE=$(git rev-parse origin/master)
  
  if [ "$LOCAL" = "$REMOTE" ]; then
    log "✅ 已是最新版本"
    return 1
  fi
  
  log "📦 发现新版本: $REMOTE"
  return 0
}

# 2. 备份当前版本
backup_current() {
  log "💾 备份当前版本..."
  
  BACKUP_NAME="chat-hub-$(date +%Y%m%d-%H%M%S).tar.gz"
  tar -czf "$BACKUP_DIR/$BACKUP_NAME" -C "$REPO_DIR" chat-hub
  
  log "✅ 备份完成: $BACKUP_NAME"
}

# 3. 拉取更新
pull_updates() {
  log "📥 拉取更新..."
  
  cd "$REPO_DIR"
  
  # 保存本地配置
  cp chat-hub/config/local.json /tmp/local.json.bak 2>/dev/null
  
  # 拉取更新
  if git pull --rebase origin master; then
    log "✅ 拉取成功"
  else
    log "❌ 拉取失败"
    return 1
  fi
  
  # 恢复本地配置
  cp /tmp/local.json.bak chat-hub/config/local.json 2>/dev/null
  
  return 0
}

# 4. 安装依赖
install_deps() {
  log "📦 安装依赖..."
  
  cd "$REPO_DIR/chat-hub"
  
  if npm install; then
    log "✅ 依赖安装完成"
  else
    log "❌ 依赖安装失败"
    return 1
  fi
  
  return 0
}

# 5. 数据库迁移（如果需要）
migrate_database() {
  log "🔄 检查数据库迁移..."
  
  # 检查是否有迁移脚本
  MIGRATION_SCRIPT="$REPO_DIR/chat-hub/migrations/latest.sql"
  
  if [ -f "$MIGRATION_SCRIPT" ]; then
    log "📝 执行数据库迁移..."
    sqlite3 "$HOME/.openclaw/chat-data/messages.db" < "$MIGRATION_SCRIPT"
    log "✅ 迁移完成"
  else
    log "ℹ️  无需迁移"
  fi
}

# 6. 重启服务
restart_service() {
  log "🔄 重启服务..."
  
  # 停止旧进程
  if [ -f "$HOME/.openclaw/chat-hub.pid" ]; then
    pid=$(cat "$HOME/.openclaw/chat-hub.pid")
    log "停止进程 (PID: $pid)"
    kill "$pid" 2>/dev/null
    sleep 3
    kill -9 "$pid" 2>/dev/null
  fi
  
  # 启动新进程
  cd "$REPO_DIR/chat-hub"
  ./start-with-log.sh
  
  sleep 5
  
  # 验证启动
  if curl -s -m 5 "http://localhost:3000/api/stats" | grep -q '"success":true'; then
    log "✅ 服务启动成功"
    return 0
  else
    log "❌ 服务启动失败"
    return 1
  fi
}

# 7. 回滚（如果启动失败）
rollback() {
  log "⚠️ 更新失败，开始回滚..."
  
  cd "$REPO_DIR"
  git reset --hard HEAD~1
  
  cd chat-hub
  npm install
  ./start-with-log.sh
  
  log "✅ 回滚完成"
}

# 主流程
main() {
  log "🚀 开始自动更新流程..."
  
  # 检查更新
  if ! check_updates; then
    exit 0
  fi
  
  # 备份
  backup_current || exit 1
  
  # 拉取更新
  if ! pull_updates; then
    log "❌ 更新失败"
    exit 1
  fi
  
  # 安装依赖
  if ! install_deps; then
    log "❌ 依赖安装失败"
    rollback
    exit 1
  fi
  
  # 数据库迁移
  migrate_database
  
  # 重启服务
  if ! restart_service; then
    log "❌ 服务启动失败"
    rollback
    exit 1
  fi
  
  log "🎉 更新成功！"
  
  # 发送通知
  send_notification "更新成功" "chat-hub 已更新到最新版本"
}

# 发送通知
send_notification() {
  title="$1"
  message="$2"
  
  curl -s -X POST "http://localhost:3000/api/reply" \
    -H "Content-Type: application/json" \
    -d "{\"content\":\"📢 **${title}**\\n\\n${message}\\n\\n—— 自动更新系统\",\"sender\":\"系统\"}" > /dev/null
}

main
```

### 方案 B：定时检查 + 自动更新

**cron 配置：**

```bash
# 每天凌晨 3 点检查更新
0 3 * * * /home/用户名/.openclaw/scripts/auto-update.sh >> /home/用户名/.openclaw/logs/auto-update.log 2>&1
```

**或使用 systemd timer：**

```ini
# ~/.config/systemd/user/chat-hub-update.service
[Unit]
Description=Auto-update chat-hub

[Service]
Type=oneshot
ExecStart=/home/用户名/.openclaw/scripts/auto-update.sh
```

```ini
# ~/.config/systemd/user/chat-hub-update.timer
[Unit]
Description=Run chat-hub auto-update daily

[Timer]
OnCalendar=*-*-* 03:00:00
Persistent=true

[Install]
WantedBy=timers.target
```

**启用：**
```bash
systemctl --user daemon-reload
systemctl --user enable chat-hub-update.timer
systemctl --user start chat-hub-update.timer
```

### 方案 C：Webhook 触发更新

**服务端添加 API：**

```javascript
// src/server.js
app.post('/api/admin/update', auth.requireAdmin, async (req, res) => {
  try {
    console.log('[Admin] 触发更新...');
    
    // 执行更新脚本
    const { exec } = require('child_process');
    exec('/home/用户名/.openclaw/scripts/auto-update.sh', (error, stdout, stderr) => {
      if (error) {
        console.error('[Admin] 更新失败:', error);
        return;
      }
      console.log('[Admin] 更新成功');
    });
    
    res.json({ success: true, message: '更新已触发' });
  } catch (error) {
    res.status(500).json({ success: false, error: error.message });
  }
});
```

**管理员触发：**

```bash
# 通过 API 触发更新
curl -X POST "http://localhost:3000/api/admin/update" \
  -H "Authorization: Bearer <admin-token>"
```

### 手动更新流程

```bash
#!/bin/bash
# manual-update.sh

echo "🔍 检查更新..."
cd ~/.openclaw/openclaw-dindin-chart
git fetch origin master

LOCAL=$(git rev-parse HEAD)
REMOTE=$(git rev-parse origin/master)

if [ "$LOCAL" = "$REMOTE" ]; then
  echo "✅ 已是最新版本"
  exit 0
fi

echo "📦 发现新版本"
echo "当前版本: $LOCAL"
echo "最新版本: $REMOTE"

read -p "是否更新？[y/N] " -n 1 -r

if [[ ! $REPLY =~ ^[Yy]$ ]]; then

cd chat-hub

echo "🔄 重启服务..."
./start-with-log.sh
```

### 批量更新（多机器人）

**批量更新脚本：**

```bash
#!/bin/bash
# batch-update.sh

BOTS=(
  "小琳:localhost"
  "小猪:172.16.30.210"
)

for bot in "${BOTS[@]}"; do
  name=$(echo "$bot" | cut -d: -f1)
  host=$(echo "$bot" | cut -d: -f2)
  
  echo "=== 更新 $name ($host) ==="
  
  if [ "$host" = "localhost" ]; then
    ~/.openclaw/scripts/auto-update.sh
  else
    ssh maple@"$host" "~/.openclaw/scripts/auto-update.sh"
  fi
  
  echo ""
done

echo "🎉 批量更新完成！"
```

### 手动回滚（指定版本）

```bash
#!/bin/bash
# rollback.sh

VERSION="$1"

if [ -z "$VERSION" ]; then
  echo "用法: ./rollback.sh <version>"
  echo "示例: ./rollback.sh v1.1.0"
  exit 1
fi

cd ~/.openclaw/openclaw-dindin-chart

echo "🔄 回滚到版本: $VERSION"

# 1. 备份当前版本
BACKUP_NAME="chat-hub-before-rollback-$(date +%Y%m%d-%H%M%S).tar.gz"
tar -czf ~/.openclaw/backups/"$BACKUP_NAME" chat-hub

# 2. 检出指定版本
git checkout "$VERSION"

# 3. 安装依赖
cd chat-hub
npm install

# 4. 重启服务
./start-with-log.sh

echo "✅ 回滚完成！"
```

### 从备份恢复

```bash
#!/bin/bash
# restore-from-backup.sh

BACKUP_FILE="$1"

if [ -z "$BACKUP_FILE" ]; then
  echo "可用备份："
  ls -lh ~/.openclaw/backups/*.tar.gz
  exit 1
fi

echo "📦 从备份恢复: $BACKUP_FILE"

# 1. 停止服务
if [ -f ~/.openclaw/chat-hub.pid ]; then
  kill $(cat ~/.openclaw/chat-hub.pid)
  sleep 3
fi

# 2. 解压备份
cd ~/.openclaw/openclaw-dindin-chart
rm -rf chat-hub.old
mv chat-hub chat-hub.old
tar -xzf "$BACKUP_FILE"

# 3. 启动服务
cd chat-hub
./start-with-log.sh

echo "✅ 恢复完成！"
```

## 总结

### 推荐方案

- 定时检查（每天凌晨 3 点）
