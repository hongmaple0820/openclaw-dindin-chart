## 崩溃场景分析

### 场景 1：端口被占用

**现象：**
```
Error: listen EADDRINUSE: address already in use :::3000
```

**原因：**
- 多个 chat-hub 进程同时启动
- 上一个进程没有正确停止
- systemd/pm2 自动重启冲突

**影响：**
- ❌ 服务无法启动
- ❌ 消息无法接收
- ❌ Redis 订阅失败

**解决方案：**
1. 启动前检查端口占用
2. 使用 PID 文件锁
3. 优雅停止旧进程

---

### 场景 2：Redis 连接失败

**现象：**
```
Error: connect ECONNREFUSED 47.96.248.176:6379
Error: Redis connection lost
```

**原因：**
- Redis 服务器宕机
- 网络不可达
- 密码错误
- 连接数超限

**影响：**
- ❌ 消息无法同步
- ❌ 多 AI 失联
- ❌ 发送失败

**解决方案：**
1. 自动重连机制（exponential backoff）
2. 降级到本地模式
3. 缓存待发送消息

---

### 场景 3：SQLite 数据库锁

**现象：**
```
SqliteError: SQLITE_BUSY: database is locked
```

**原因：**
- 多个进程同时写入
- 事务未提交
- 长时间读取锁定

**影响：**
- ❌ 消息存储失败
- ❌ 数据不一致

**解决方案：**
1. WAL 模式（Write-Ahead Logging）
2. 重试机制（3 次，间隔递增）
3. 超时设置

### 场景 4：钉钉 Webhook 失败

**现象：**
```
Error: Request failed with status code 400
Error: sign not match
```

**原因：**
- Webhook URL 错误
- Secret 错误
- 签名算法错误
- 钉钉限流（20 条/分钟）

**影响：**
- ❌ 消息发送到钉钉失败
- ❌ 人类看不到回复

**解决方案：**
1. 重试机制（3 次）
2. 降级通知（日志记录）
3. 频率限制

### 场景 5：OpenClaw 触发失败

**现象：**
```
Error: spawn openclaw ENOENT
Error: Command failed with exit code 1
```

**原因：**
- openclaw 不在 PATH
- Gateway 未启动
- 权限不足

**影响：**
- ❌ AI 无法回复
- ❌ 消息堆积

**解决方案：**
1. 检查 openclaw 可用性
2. 降级到手动处理
3. 告警通知

### 场景 6：内存泄漏/OOM

**现象：**
```
JavaScript heap out of memory
Killed (OOM killer)
```

**原因：**
- 消息历史过多
- 订阅者未清理
- Redis 订阅消息堆积

**影响：**
- ❌ 进程崩溃
- ❌ 服务不可用

**解决方案：**
1. 限制内存使用（--max-old-space-size）
2. 定期清理历史消息
3. 监控内存使用

### 1. 进程级防护

#### PID 文件锁

```javascript
// src/utils/pid-lock.js
const fs = require('fs');
const path = require('path');

class PidLock {
  constructor(pidFile) {
    this.pidFile = pidFile;
  }

  // 获取锁
  async acquire() {
    // 检查 PID 文件是否存在
    if (fs.existsSync(this.pidFile)) {
      const oldPid = fs.readFileSync(this.pidFile, 'utf8').trim();
      
      // 检查进程是否还在运行
      try {
        process.kill(oldPid, 0); // 信号 0 只检测，不杀死
        throw new Error(`Another instance is running (PID: ${oldPid})`);
      } catch (e) {
        if (e.code === 'ESRCH') {
          // 进程不存在，清理旧 PID 文件
          console.log(`[PidLock] Stale PID file found, cleaning up`);
          fs.unlinkSync(this.pidFile);
        } else {
          throw e;
        }
      }
    }

    // 写入当前 PID
    fs.writeFileSync(this.pidFile, process.pid.toString());
    console.log(`[PidLock] Acquired (PID: ${process.pid})`);

    // 退出时清理
    process.on('exit', () => this.release());
    process.on('SIGINT', () => { this.release(); process.exit(); });
    process.on('SIGTERM', () => { this.release(); process.exit(); });
  }

  // 释放锁
  release() {
    if (fs.existsSync(this.pidFile)) {
      fs.unlinkSync(this.pidFile);
      console.log(`[PidLock] Released`);
    }
  }
}

module.exports = PidLock;
```

**使用：**
```javascript
// src/index.js
const PidLock = require('./utils/pid-lock');

async function main() {
  const pidLock = new PidLock('~/.openclaw/chat-hub.pid');
  
  try {
    await pidLock.acquire();
    // 启动服务
  } catch (error) {
    console.error('Failed to acquire PID lock:', error.message);
    process.exit(1);
  }
}
```

#### 自动重连 + 降级

```javascript
// src/redis-client.js
const Redis = require('ioredis');

class ResilientRedisClient {
  constructor(config) {
    this.config = config;
    this.client = null;
    this.reconnecting = false;
    this.fallbackMode = false;
    this.pendingMessages = []; // 降级时缓存
  }

  async connect() {
    this.client = new Redis({
      host: this.config.host,
      port: this.config.port,
      password: this.config.password,
      retryStrategy: (times) => {
        const delay = Math.min(times * 1000, 30000); // 最多 30 秒
        console.log(`[Redis] Reconnecting in ${delay}ms (attempt ${times})`);
        return delay;
      },
      maxRetriesPerRequest: 3,
      enableReadyCheck: true,
    });

    this.client.on('connect', () => {
      console.log('[Redis] Connected');
      this.reconnecting = false;
      this.fallbackMode = false;
      this.flushPendingMessages(); // 发送缓存的消息
    });

    this.client.on('error', (err) => {
      console.error('[Redis] Error:', err.message);
      if (!this.reconnecting) {
        this.enterFallbackMode();
      }
    });

    this.client.on('close', () => {
      console.warn('[Redis] Connection closed');
      this.reconnecting = true;
    });

    return this.client;
  }

  // 降级到本地模式
  enterFallbackMode() {
    console.warn('[Redis] Entering fallback mode (local only)');
    this.fallbackMode = true;
  }

  // 发送消息（带降级）
  async publish(channel, message) {
    if (this.fallbackMode) {
      console.warn(`[Redis] Fallback: caching message for ${channel}`);
      this.pendingMessages.push({ channel, message });
      return;
    }

    try {
      await this.client.publish(channel, message);
    } catch (error) {
      console.error('[Redis] Publish failed:', error.message);
      this.enterFallbackMode();
      this.pendingMessages.push({ channel, message });
    }
  }

  // 恢复后发送缓存的消息
  async flushPendingMessages() {
    if (this.pendingMessages.length === 0) return;

    console.log(`[Redis] Flushing ${this.pendingMessages.length} pending messages`);
    
    for (const { channel, message } of this.pendingMessages) {
      try {
        await this.client.publish(channel, message);
      } catch (error) {
        console.error('[Redis] Failed to flush message:', error.message);
      }
    }

    this.pendingMessages = [];
  }
}

module.exports = ResilientRedisClient;
```

#### WAL 模式 + 重试

```javascript
// src/message-store.js
const Database = require('better-sqlite3');

class MessageStore {
  constructor(dbPath) {
    this.db = new Database(dbPath, {
      timeout: 10000, // 10 秒超时
    });

    // 启用 WAL 模式（写前日志，避免锁冲突）
    this.db.pragma('journal_mode = WAL');
    this.db.pragma('busy_timeout = 10000');

    // 初始化表
    this.initTables();
  }

  // 重试包装器
  async withRetry(fn, maxRetries = 3) {
    for (let i = 0; i < maxRetries; i++) {
      try {
        return fn();
      } catch (error) {
        if (error.code === 'SQLITE_BUSY' && i < maxRetries - 1) {
          const delay = Math.pow(2, i) * 100; // 100ms, 200ms, 400ms
          console.warn(`[SQLite] Busy, retrying in ${delay}ms (${i + 1}/${maxRetries})`);
          await new Promise(resolve => setTimeout(resolve, delay));
        } else {
          throw error;
        }
      }
    }
  }

  // 保存消息（带重试）
  async saveMessage(message) {
    return this.withRetry(() => {
      const stmt = this.db.prepare(`
        INSERT INTO messages (id, type, sender, content, timestamp, source)
        VALUES (?, ?, ?, ?, ?, ?)
      `);
      stmt.run(
        message.id,
        message.type,
        message.sender,
        message.content,
        message.timestamp,
        message.source
      );
    });
  }
}

module.exports = MessageStore;
```

```javascript
const { promisify } = require('util');
const execAsync = promisify(exec);

class OpenClawTrigger {
  constructor(config) {
    this.config = config;
    this.available = false;
    this.lastCheck = 0;
  }

  // 健康检查
  async healthCheck() {
    const now = Date.now();
    if (now - this.lastCheck < 60000) { // 缓存 1 分钟
      return this.available;
    }

    try {
      const { stdout } = await execAsync('which openclaw', { timeout: 5000 });
      this.available = !!stdout.trim();
      this.lastCheck = now;
      console.log('[OpenClaw] Health check: OK');
      return true;
    } catch (error) {
      this.available = false;
      this.lastCheck = now;
      console.warn('[OpenClaw] Health check: FAILED');
      return false;
    }
  }

  // 触发（带降级）
  async trigger(message) {
    // 先检查健康状态
    const healthy = await this.healthCheck();
    
    if (!healthy) {
      console.warn('[OpenClaw] Not available, skipping trigger');
      return false;
    }

    try {
      const cmd = `openclaw system event --text "${message}" --mode now`;
      await execAsync(cmd, { timeout: 30000 });
      console.log('[OpenClaw] Trigger success');
      return true;
    } catch (error) {
      console.error('[OpenClaw] Trigger failed:', error.message);
      return false;
    }
  }
}
```

```javascript
// src/utils/memory-guard.js
class MemoryGuard {
  constructor(maxMemoryMB = 512) {
    this.maxMemoryMB = maxMemoryMB;
    this.checkIntervalMs = 60000; // 每分钟检查
    this.startMonitoring();
  }

  startMonitoring() {
    setInterval(() => {
      const usage = process.memoryUsage();
      const usedMB = Math.round(usage.heapUsed / 1024 / 1024);
      
      console.log(`[Memory] Used: ${usedMB}MB / ${this.maxMemoryMB}MB`);

      if (usedMB > this.maxMemoryMB * 0.9) {
        console.warn('[Memory] High usage, triggering GC');
        if (global.gc) {
          global.gc();
        }
      }

      if (usedMB > this.maxMemoryMB) {
        console.error('[Memory] Limit exceeded, exiting gracefully');
        process.exit(1); // 让监控脚本重启
      }
    }, this.checkIntervalMs);
  }
}

module.exports = MemoryGuard;
```

### 方案 2：监控脚本自动恢复

**增强版监控脚本：**
```bash
#!/bin/bash
# monitor-chat-hub-enhanced.sh

PID_FILE="$HOME/.openclaw/chat-hub.pid"
LOG_FILE="$HOME/.openclaw/logs/chat-hub-monitor.log"
FAIL_COUNT_FILE="$HOME/.openclaw/chat-hub-fail-count"
MAX_FAILS=3

log() {
  echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a "$LOG_FILE"
}

# 检查进程
check_process() {
  if [ ! -f "$PID_FILE" ]; then
    return 1
  fi

  pid=$(cat "$PID_FILE")
  if ! kill -0 "$pid" 2>/dev/null; then
    return 1
  fi

  return 0
}

# 检查 API
check_api() {
  response=$(curl -s -m 5 "http://localhost:3000/api/stats" 2>/dev/null)
  
  if echo "$response" | grep -q '"success":true'; then
    return 0
  fi

  return 1
}

# 重启服务
restart_service() {
  log "⚠️ 检测到故障，准备重启..."

  # 停止旧进程
  if [ -f "$PID_FILE" ]; then
    pid=$(cat "$PID_FILE")
    log "停止旧进程 (PID: $pid)"
    kill "$pid" 2>/dev/null
    sleep 2
    kill -9 "$pid" 2>/dev/null
  fi

  # 启动新进程
  cd "$HOME/.openclaw/openclaw-dindin-chart/chat-hub"
  ./start-with-log.sh
  
  sleep 5

  if check_api; then
    log "✅ 重启成功"
    echo "0" > "$FAIL_COUNT_FILE"
    return 0
  else
    log "❌ 重启失败"
    return 1
  fi
}

# 主逻辑
main() {
  # 检查进程
  if ! check_process; then
    log "❌ 进程不存在"
    restart_service
    exit 0
  fi

  # 检查 API（递增等待）
  for i in 1 2 3; do
    if check_api; then
      log "✅ 服务正常 (尝试 $i/3)"
      echo "0" > "$FAIL_COUNT_FILE"
      exit 0
    fi

    wait_time=$((30 + i * 10))
    log "⚠️ API 无响应 (尝试 $i/3)，等待 ${wait_time}s..."
    sleep "$wait_time"
  done

  # 三次失败，重启
  fail_count=$(cat "$FAIL_COUNT_FILE" 2>/dev/null || echo "0")
  fail_count=$((fail_count + 1))
  echo "$fail_count" > "$FAIL_COUNT_FILE"

  log "❌ 服务异常 (连续失败: $fail_count)"

  if [ "$fail_count" -ge "$MAX_FAILS" ]; then
    log "🚨 达到最大失败次数，触发告警"
    # TODO: 发送告警（钉钉/邮件）
  fi

  restart_service
}

main
```

## 总结

### 防护层级

```
┌─────────────────────────────────┐
│   L1: 进程锁（PID 文件）         │
├─────────────────────────────────┤
│   L2: 连接防护（重试 + 降级）    │
├─────────────────────────────────┤
│   L3: 数据库防护（WAL + 重试）   │
├─────────────────────────────────┤
│   L4: 业务防护（频率限制 + 循环） │
├─────────────────────────────────┤
│   L5: 资源防护（内存限制 + GC）   │
├─────────────────────────────────┤
│   L6: 监控恢复（自动重启 + 告警） │
└─────────────────────────────────┘
```

### 关键原则

1. **多层防护** - 不依赖单一机制
2. **优雅降级** - 部分故障不影响全局
3. **快速恢复** - 自动化重启机制
4. **主动监控** - 及时发现问题
5. **持续改进** - 根据实际情况调整
