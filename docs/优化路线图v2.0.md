#### 1.1 PID 文件锁

**文件：** `src/utils/pid-lock.js`

```javascript
const fs = require('fs');
const path = require('path');
const os = require('os');

class PidLock {
  constructor(pidFile) {
    this.pidFile = pidFile || path.join(os.homedir(), '.openclaw/chat-hub.pid');
  }

  async acquire() {
    // 检查 PID 文件是否存在
    if (fs.existsSync(this.pidFile)) {
      const oldPid = parseInt(fs.readFileSync(this.pidFile, 'utf8').trim());
      
      // 检查进程是否还在运行
      try {
        process.kill(oldPid, 0); // 信号 0 只检测，不杀死
        throw new Error(`另一个实例正在运行 (PID: ${oldPid})`);
      } catch (e) {
        if (e.code === 'ESRCH') {
          // 进程不存在，清理旧 PID 文件
          console.log(`[PidLock] 清理陈旧 PID 文件`);
          fs.unlinkSync(this.pidFile);
        } else {
          throw e;
        }
      }
    }

    // 写入当前 PID
    const pidDir = path.dirname(this.pidFile);
    if (!fs.existsSync(pidDir)) {
      fs.mkdirSync(pidDir, { recursive: true });
    }
    fs.writeFileSync(this.pidFile, process.pid.toString());
    console.log(`[PidLock] 已获取锁 (PID: ${process.pid})`);

    // 退出时清理
    const cleanup = () => this.release();
    process.on('exit', cleanup);
    process.on('SIGINT', () => { this.release(); process.exit(); });
    process.on('SIGTERM', () => { this.release(); process.exit(); });
  }

  release() {
    if (fs.existsSync(this.pidFile)) {
      fs.unlinkSync(this.pidFile);
      console.log(`[PidLock] 已释放锁`);
    }
  }
}

module.exports = PidLock;
```

#### 1.2 SQLite 重试包装器

```javascript
class MessageStore {
  constructor(dbPath) {
    this.db = new Database(dbPath, {
      timeout: 10000, // 10 秒超时
    });

    // 启用 WAL 模式（写前日志，避免锁冲突）
    this.db.pragma('journal_mode = WAL');
    this.db.pragma('busy_timeout = 10000');

    this.initTables();
  }

  // 重试包装器
  async withRetry(fn, maxRetries = 3) {
    for (let i = 0; i < maxRetries; i++) {
      try {
        return fn();
      } catch (error) {
        if (error.code === 'SQLITE_BUSY' && i < maxRetries - 1) {
          const delay = Math.pow(2, i) * 100; // 100ms, 200ms, 400ms
          console.warn(`[SQLite] 数据库忙，${delay}ms 后重试 (${i + 1}/${maxRetries})`);
          await new Promise(resolve => setTimeout(resolve, delay));
        } else {
          throw error;
        }
      }
    }
  }

  // 保存消息（带重试）
  async saveMessage(message) {
    return this.withRetry(() => {
      const stmt = this.db.prepare(`
        INSERT OR REPLACE INTO messages (id, type, sender, content, timestamp, source, atTargets, replyTo)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?)
      `);
      stmt.run(
        message.id,
        message.type,
        message.sender,
        message.content,
        message.timestamp,
        message.source,
        JSON.stringify(message.atTargets),
        message.replyTo
      );
      return message;
    });
  }
}
```

```javascript
const Redis = require('ioredis');

class ResilientRedisClient {
  constructor(config) {
    this.config = config;
    this.client = null;
    this.subscriber = null;
    this.publisher = null;
    this.fallbackMode = false;
    this.pendingMessages = []; // 降级时缓存
  }

  async connect() {
    const redisConfig = {
      host: this.config.host,
      port: this.config.port,
      password: this.config.password,
      retryStrategy: (times) => {
        const delay = Math.min(times * 1000, 30000); // 最多 30 秒
        console.log(`[Redis] ${delay}ms 后重连 (尝试 ${times})`);
        return delay;
      },
      maxRetriesPerRequest: 3,
      enableReadyCheck: true,
      lazyConnect: false,
    };

    // 创建三个连接
    this.client = new Redis(redisConfig);
    this.subscriber = new Redis(redisConfig);
    this.publisher = new Redis(redisConfig);

    // 事件监听
    this.client.on('connect', () => {
      console.log('[Redis] 客户端已连接');
      this.fallbackMode = false;
      this.flushPendingMessages();
    });

    this.client.on('error', (err) => {
      console.error('[Redis] 客户端错误:', err.message);
      this.enterFallbackMode();
    });

    this.subscriber.on('connect', () => {
      console.log('[Redis] 订阅者已连接');
    });

    this.publisher.on('connect', () => {
      console.log('[Redis] 发布者已连接');
    });

    // 等待连接就绪
    await Promise.all([
      this.client.ping(),
      this.subscriber.ping(),
      this.publisher.ping(),
    ]);

    console.log('[Redis] 所有连接就绪');
    return this;
  }

  // 降级到本地模式
  enterFallbackMode() {
    if (!this.fallbackMode) {
      console.warn('[Redis] 🚨 进入降级模式（仅本地）');
      this.fallbackMode = true;
    }
  }

  // 发送消息（带降级）
  async publish(channel, message) {
    if (this.fallbackMode) {
      console.warn(`[Redis] 降级：缓存消息 (${channel})`);
      this.pendingMessages.push({ channel, message, timestamp: Date.now() });
      
      // 限制缓存数量
      if (this.pendingMessages.length > 1000) {
        console.warn('[Redis] 缓存消息过多，丢弃最旧的');
        this.pendingMessages.shift();
      }
      return;
    }

    try {
      await this.publisher.publish(channel, message);
    } catch (error) {
      console.error('[Redis] 发布失败:', error.message);
      this.enterFallbackMode();
      this.pendingMessages.push({ channel, message, timestamp: Date.now() });
    }
  }

  // 恢复后发送缓存的消息
  async flushPendingMessages() {
    if (this.pendingMessages.length === 0) return;

    console.log(`[Redis] 📤 发送 ${this.pendingMessages.length} 条缓存消息`);
    
    for (const { channel, message } of this.pendingMessages) {
      try {
        await this.publisher.publish(channel, message);
      } catch (error) {
        console.error('[Redis] 发送缓存消息失败:', error.message);
      }
    }

    this.pendingMessages = [];
    console.log('[Redis] ✅ 缓存消息已清空');
  }

  // 订阅频道
  async subscribe(channel, handler) {
    await this.subscriber.subscribe(channel);
    this.subscriber.on('message', (ch, msg) => {
      if (ch === channel) {
        handler(msg);
      }
    });
    console.log(`[Redis] 已订阅频道: ${channel}`);
  }
}

module.exports = ResilientRedisClient;
```

```javascript
class SSEManager {
  constructor(redisClient) {
    this.clients = new Map(); // userId -> { res, lastPing }
    this.redisClient = redisClient;
    
    // 心跳检测（每30秒）
    setInterval(() => this.heartbeat(), 30000);
    
    // 监听 Redis 消息
    this.subscribeToMessages();
  }

  // 新客户端连接
  async connect(req, res, userId) {
    // 设置 SSE 头
    res.writeHead(200, {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache',
      'Connection': 'keep-alive',
      'Access-Control-Allow-Origin': '*',
    });

    // 保存连接
    this.clients.set(userId, { res, lastPing: Date.now() });
    console.log(`[SSE] 用户 ${userId} 已连接 (总计 ${this.clients.size})`);

    // 发送欢迎消息
    this.send(userId, 'connected', { userId, timestamp: Date.now() });

    // 通知其他用户（某人上线）
    this.broadcast('user-online', { userId, timestamp: Date.now() }, userId);

    // 更新在线列表
    await this.updateOnlineUsers();

    // 连接断开时清理
    req.on('close', async () => {
      this.clients.delete(userId);
      console.log(`[SSE] 用户 ${userId} 已断开 (剩余 ${this.clients.size})`);
      
      // 通知其他用户（某人下线）
      this.broadcast('user-offline', { userId, timestamp: Date.now() });
      
      // 更新在线列表
      await this.updateOnlineUsers();
    });
  }

  // 发送消息给特定用户
  send(userId, event, data) {
    const client = this.clients.get(userId);
    if (!client) return false;

    try {
      client.res.write(`event: ${event}\n`);
      client.res.write(`data: ${JSON.stringify(data)}\n\n`);
      return true;
    } catch (error) {
      console.error(`[SSE] 发送失败 (${userId}):`, error.message);
      this.clients.delete(userId);
      return false;
    }
  }

  // 广播消息给所有用户
  broadcast(event, data, excludeUserId = null) {
    let success = 0;
    for (const [userId, client] of this.clients.entries()) {
      if (userId === excludeUserId) continue;
      if (this.send(userId, event, data)) {
        success++;
      }
    }
    return success;
  }

  // 心跳检测
  heartbeat() {
    const now = Date.now();
    const timeout = 60000; // 60 秒超时

    for (const [userId, client] of this.clients.entries()) {
      // 发送 ping
      try {
        client.res.write(': ping\n\n');
        client.lastPing = now;
      } catch (error) {
        // 连接已断开
        this.clients.delete(userId);
        console.log(`[SSE] 清理断开的连接: ${userId}`);
      }
    }
  }

  // 订阅 Redis 消息
  async subscribeToMessages() {
    // 订阅新消息
    await this.redisClient.subscribe('chat:messages', (msg) => {
      const message = JSON.parse(msg);
      this.broadcast('new-message', message);
    });

    // 订阅回复消息
    await this.redisClient.subscribe('chat:replies', (msg) => {
      const message = JSON.parse(msg);
      this.broadcast('new-reply', message);
    });

    console.log('[SSE] 已订阅 Redis 频道');
  }

  // 更新在线用户列表
  async updateOnlineUsers() {
    const onlineUsers = Array.from(this.clients.keys());
    
    // 存储到 Redis
    await this.redisClient.client.del('online:users');
    if (onlineUsers.length > 0) {
      await this.redisClient.client.sadd('online:users', ...onlineUsers);
    }

    // 广播在线列表
    this.broadcast('online-users', { users: onlineUsers, count: onlineUsers.length });
  }

  // 获取在线用户列表
  async getOnlineUsers() {
    const users = await this.redisClient.client.smembers('online:users');
    return users || [];
  }
}

module.exports = SSEManager;
```

```javascript
class SSEClient {
  constructor(userId) {
    this.userId = userId;
    this.eventSource = null;
    this.handlers = new Map();
    this.reconnectDelay = 1000;
    this.maxReconnectDelay = 30000;
  }

  connect() {
    const url = `${import.meta.env.VITE_API_BASE}/api/sse/${this.userId}`;
    
    this.eventSource = new EventSource(url);

    // 连接成功
    this.eventSource.addEventListener('connected', (e) => {
      console.log('[SSE] 已连接:', JSON.parse(e.data));
      this.reconnectDelay = 1000; // 重置重连延迟
      this.trigger('connected', JSON.parse(e.data));
    });

    // 新消息
    this.eventSource.addEventListener('new-message', (e) => {
      const message = JSON.parse(e.data);
      this.trigger('message', message);
    });

    // 新回复
    this.eventSource.addEventListener('new-reply', (e) => {
      const message = JSON.parse(e.data);
      this.trigger('reply', message);
    });

    // 在线用户更新
    this.eventSource.addEventListener('online-users', (e) => {
      const data = JSON.parse(e.data);
      this.trigger('online', data);
    });

    // 用户上线
    this.eventSource.addEventListener('user-online', (e) => {
      const data = JSON.parse(e.data);
      this.trigger('user-online', data);
    });

    // 用户下线
    this.eventSource.addEventListener('user-offline', (e) => {
      const data = JSON.parse(e.data);
      this.trigger('user-offline', data);
    });

    // 错误处理
    this.eventSource.onerror = (error) => {
      console.error('[SSE] 连接错误:', error);
      this.eventSource.close();
      
      // 自动重连
      setTimeout(() => {
        console.log(`[SSE] ${this.reconnectDelay}ms 后重连...`);
        this.connect();
        this.reconnectDelay = Math.min(this.reconnectDelay * 2, this.maxReconnectDelay);
      }, this.reconnectDelay);
    };
  }

  // 监听事件
  on(event, handler) {
    if (!this.handlers.has(event)) {
      this.handlers.set(event, []);
    }
    this.handlers.get(event).push(handler);
  }

  // 触发事件
  trigger(event, data) {
    const handlers = this.handlers.get(event);
    if (handlers) {
      handlers.forEach(handler => handler(data));
    }
  }

  // 断开连接
  disconnect() {
    if (this.eventSource) {
      this.eventSource.close();
      this.eventSource = null;
    }
  }
}

export default SSEClient;
```
